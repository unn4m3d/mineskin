def composite_texture!(tex, x, y, old_unit, op: Magick::SrcOverCompositeOp)
  @image.composite!(
    resize(tex.texture, old_unit),
    x * @unit,
    y * @unit,
    op
  )
  @image.composite!(
    resize(tex.overlay, old_unit),
    x * @unit,
    y * @unit,
    op
  ) if tex.overlay
end
