def crop(texture)
  @image.crop(
    texture[:x] * @unit,
    texture[:y] * @unit,
    texture[:width] * @unit,
    texture[:height] * @unit
  )
end

def extract(hash)
  Cuboid.new(
    part(hash[:top]),
    part(hash[:bottom]),
    part(hash[:left]),
    part(hash[:right]),
    part(hash[:front]),
    part(hash[:back])
  )
end

def part(texture: { x: 0, y: 0, width: 0, height: 0 }, overlay: nil)
  tex = crop coords_to_h texture

  part = Texture.new(tex, nil)
  if overlay
    over = crop coords_to_h overlay
    part.overlay = over
  end

  part
end
