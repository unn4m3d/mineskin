def render(width, background: 'white')
  @unit = image_unit size: width, count: 12
  @image = Magick::Image.new(width, 5 * width / 6) do
    self.background_color = background
  end
  render_head!
  render_body!
  render_legs!
  render_arms!
  @image
end
