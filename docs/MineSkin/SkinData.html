def initialize(filename)
  @image = Magick::Image.read(filename).first
  @new_format = @image.columns == @image.rows
  @unit = image_unit size: @image.columns
  @head = extract_head
  @body = extract_body
  @left_leg, @right_leg = extract_legs
  @left_arm, @right_arm = extract_arms
end
